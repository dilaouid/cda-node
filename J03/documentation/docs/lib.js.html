/**
 * autre paramètre de jsdocs
 * @param {number} a 
 * @param {number} b 
 * @throws {Error} déclenche une erreur si les paramètres a ou b sont négatif
 * @returns {number}
 */
function calculPrix ( a, b ){
    if(a < 0 || b < 0){
        throw new Error("a et b  doivent être positifs")
    }
    return a * b
}
